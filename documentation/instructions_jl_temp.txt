NAVIGATE TO WORKSPACE OF ROS (OR CREATE IT)
cd ~/workspace/ros

CREATE A FOLDER FOR THE WHOLE STACK APPLICATION
mkdir quadrotor_stack_catkin
cd quadrotor_stack_catkin

CREATION OF THE CATKING WORKSPACE
source /opt/ros/indigo/setup.bash
mkdir src
cd src
catkin_init_workspace
cd ..
catkin_make


source devel/setup.bash


DOWNLOAD PACKAGES
cd src

1- EXT_STACK
mkdir extStack
a. ardrone_autonomy
git clone https://github.com/AutonomyLab/ardrone_autonomy.git --branch=indigo-devel ./extStack/ardrone_autonomy
b. px-ros-pkg
git clone https://github.com/cvg/px-ros-pkg --branch=master ./extStack/px-ros-pkg


2- EXT_STACK_CVG
mkdir extStackCVG
a. okto_driver
git clone https://bitbucket.org/Vision4UAV/okto_driver.git --branch=catkin ./extStackCVG/okto_driver
b. mav_tools
git clone https://bitbucket.org/joselusl/mav_tools.git ./extStackCVG/mav_tools
c. asctec_drivers
git clone https://bitbucket.org/joselusl/asctec_drivers.git ./extStackCVG/asctec_drivers


3- STACK
git clone https://bitbucket.org/joselusl/quadrotor_swarm.git --branch=catkin ./stack



COMPILE
cd ..
catkin_make




OPENING with qtcreator
rm CMakeLists.txt
cp /opt/ros/indigo/share/catkin/cmake/toplevel.cmake CMakeLists.txt
